
CREATEPROCEDURE [dbo].[CRUDOPERATION_SP]
    @WhmCode INT,
    @Status CHAR(1),
    @Filter NVARCHAR(100),
	@EMPLOYEE_ID INT = NULL,
    @EMPLOYEE_NAME NVARCHAR(100) = NULL,
    @DESIGNATION NVARCHAR(100) = NULL,
    @POST NVARCHAR(100) = NULL,
    @PLACE NVARCHAR(100) = NULL,
    @PHONE NVARCHAR(100) = NULL
AS
BEGIN
     IF (@status = 'P')
	Begin
    SELECT EMPLOYEE_ID, EMPLOYEE_NAME, DESIGNATION, POST, PLACE, PHONE
    FROM CRUD_OPERATION_TABLE
	end
	 ELSE IF (@Status = 'I')
    BEGIN
        INSERT INTO CRUD_OPERATION_TABLE (EMPLOYEE_ID, EMPLOYEE_NAME, DESIGNATION, POST, PLACE, PHONE)
        VALUES (@EMPLOYEE_ID, @EMPLOYEE_NAME, @DESIGNATION, @POST, @PLACE, @PHONE)
    END
   
END;
